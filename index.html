1. Sealed Class

Sealed class adalah kelas yang secara eksplisit membatasi kelas-kelas mana saja yang boleh menjadi turunannya. Dengan kata lain, hierarki pewarisan kelas ini sudah ditentukan dan terbatas, tidak bisa sembarangan ditambah oleh kelas lain di luar izin. 


// Nama  : PUTRA DHAFIR NASHRULLAH
// NIM   : 251410009
// Kelas : SI2A

// Sealed class membatasi pewarisan hanya dalam satu file
sealed class ApiResponse {}

class Success extends ApiResponse {
  final String message;
  Success(this.message);
}

class Error extends ApiResponse {
  final String errorMessage;
  Error(this.errorMessage);
}

void handleResponse(ApiResponse response) {
  if (response is Success) {
    print("✅ Success: ${response.message}");
  } else if (response is Error) {
    print("❌ Error: ${response.errorMessage}");
  }
}

void main() {
  print("=" * 50);
  print("CONTOH SEALED CLASS");
  print("=" * 50);
  
  handleResponse(Success("Data berhasil diambil"));
  handleResponse(Error("Terjadi kesalahan pada server"));
}
Jenis-jenis Constructor
  1. Default Constructor: ClassName()
  2. Named Constructor: ClassName.name()
  3. Parameterized Constructor: ClassName(this.param)
  4. Constant Constructor: const ClassName()
  5. Factory Constructor (lanjut)

2. CLASS MOBIL

Constant constructor adalah jawaban Dart atas kebutuhan akan objek yang immutable — objek yang sekali diciptakan, takkan pernah berubah selamanya. Dengan menambahkan kata kunci const, kita memberi "sumpah" pada objek untuk tidak mengubah identitasnya. Semua propertinya pun harus final, seperti takdir yang sudah ditetapkan.

Dari sudut pandang komputasi, constant constructor memungkinkan canonicalization objek-objek dengan nilai yang sama akan menempati lokasi memori yang sama, menghemat sumber daya. Ini seperti konsep dalam pemrograman—objek-objek yang serupa pada hakikatnya adalah satu entitas yang sama. 


// Nama        : Putra Dhafir Nashrullah
// NIM         : 251410009
// Kelas       : SISTEM INFORMASI SI2A

import 'dart:convert';
class Mobil {
  String? merk;
  int? tahun;

  // Parameterized Constructor
  Mobil(this.merk, this.tahun);
  
  // Named Constructor
  Mobil.baru() {
    merk = 'ayla';
    tahun = 2025;
  }

  // Named Constructor dariJson
  Mobil.dariJson(Map json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  @override
  String toString() {
    return 'Mobil{merk: $merk, tahun: $tahun}';
  }
}

// Fungsi main() wajib ada untuk menjalankan program
void main() {
  // Membuat objek dengan Parameterized Constructor
  Mobil mobil1 = Mobil('toyota', 2024);
  print('Mobil 1: $mobil1');

  // Membuat objek dengan Named Constructor baru()
  Mobil mobil2 = Mobil.baru();
  print('Mobil 2: $mobil2');

  // Membuat objek dengan Named Constructor dariJson()
  Map dataMobil = {'merk': 'honda', 'tahun': 2021};
  Mobil mobil3 = Mobil.dariJson(dataMobil);
  print('Mobil 3: $mobil3');
}

3. CLASS WARNA

Objek const dengan nilai berbeda akan berbeda instance. 


// Nama        : Putra Dhafir Nashrullah
// NIM         : 251410009
// Kelas       : SISTEM INFORMASI SI2A
  class Warna {
  final int r, g, b;
  const Warna(this.r, this.g, this.b);

  @override
  String toString() {
    return 'Warna(r: $r, g: $g, b: $b)';
  }
}

void main() {
  const merah = Warna(255, 0, 0);
  const merahLagi = Warna(255, 0, 0); // Menggunakan instance yang sama di memory

  print(merah);
  print(merahLagi);
  
  // Membuktikan bahwa instance-nya sama (canonicalized)
  print(identical(merah, merahLagi)); // true
}

4. CLASS TITIK

Initializer list adalah ruang persiapan sebelum constructor body dieksekusi—seperti fase emantenatal dalam siklus hidup objek. Ditandai dengan tanda titik dua (:) setelah parameter constructor, ia memungkinkan kita memanipulasi nilai sebelum objek benar-benar "dilahirkan" ke alam memori. 


// Nama        : Putra Dhafir Nashrullah
// NIM         : 251410009
// Kelas       : SISTEM INFORMASI SI2A
  class Titik {
  final double x, y;

  Titik(double x, double y)
    : x = x.abs(),
      y = y.abs() {
    print('Titik dibuat di (${this.x}, ${this.y})');
  }
}

void main(){
  // Contoh penggunaan class Titik
  print('Membuat objek Titik pertama:');
  var titik1 = Titik(3.0, 4.0);
  print('Titik 1: x = ${titik1.x}, y = ${titik1.y}');

  print('\nMembuat objek Titik kedua dengan nilai negatif:');
  var titik2 = Titik(-5.0, 10.0);
  print('Titik 2: x = ${titik2.x}, y = ${titik2.y}');

  print('\nMembuat objek Titik ketiga dengan nilai negatif di kedua koordinat:');
  var titik3 = Titik(-1.5, -2.5);
  print('Titik 3: x = ${titik3.x}, y = ${titik3.y}');
}

}

void main() {

  var m2 = Mahasiswa.baru("Budi", 19);

  var m4 = Mahasiswa.fromJson({
    'nama': 'PUTRA',
    'umur': 18,
    'nim': '251410009'
  });
  
  m1.tampilkan();
  m2.tampilkan();
  m3.tampilkan();

7. CLASS GETTER DAN SETTER

Getter dan Setter adalah method dalam Pemrograman Berorientasi Objek (OOP) yang digunakan untuk mengambil (getter) dan mengubah (setter) nilai dari sebuah variabel dalam class. Biasanya digunakan bersama dengan konsep enkapsulasi untuk mengontrol akses terhadap data. 


// Nama        : Putra Dhafir Nashrullah
// NIM         : 251410009
// Kelas       : SISTEM INFORMASI SI2A
class Person{
  String?_name;
  int?age;

  Person(this._name,this.age);

  // Getter (konsistenkan nama untuk kemudahan pemakaian)
  String?get nama => _name;
  String?getName() => _name;

  // Setter (konsistenkan nama)
  set nama(String?value) => _name = value;
  void setName(String?value) => _name = value;
}

// Tambahkan fungsi main() untuk menjalankan program
void main() {
  // Membuat objek Person
  Person orang = Person("Putra", 18);
  
  // Menggunakan getter
  print("Nama awal: ${orang.nama}");
  print("Nama via getName(): ${orang.getName()}");
  
  // Menggunakan setter
  orang.nama = "Dhafir";
  print("Nama setelah diubah: ${orang.nama}");
  
  orang.setName("Nashrullah");
  print("Nama via setName(): ${orang.getName()}");
}

8. CLASS PRODUK VALIDASI

Setelah memahami teori, mari kita lihat enkapsulasi dalam skenario nyata: sistem login. Di sini, data sensitif seperti username dan password disembunyikan di balik field privat. Tidak ada akses langsung dari luar—semua interaksi harus melalui method publik yang sudah divalidasi. 


// Nama  : PUTRA DHAFIR NASHRULLAH
// Nim   : 251410009
// Kelas : SI2A

class Produk {
  String _nama;
  double _harga;

  Produk(this._nama, this._harga);

  String get nama => _nama;
  double get harga => _harga;

  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('Gagal memperbarui nama: Nama harus memiliki minimal 3 karakter.');
    }
  }

  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('Gagal memperbarui harga: Harga harus lebih besar dari 0.');
    }
  }
}

void main() {
  var produk1 = Produk('Monitor Samsung', 3500.0);

  print('--- Informasi Produk Awal ---');
  print('Nama Produk: ${produk1.nama}');
  print('Harga Produk: Rp ${produk1.harga}');

  print('\n--- Memperbarui Produk ---');
  produk1.nama = 'Keyboard Mekanik';
  produk1.harga = 1200.0;

  print('Nama (setelah update): ${produk1.nama}');

  print('\n--- Menguji Validasi Setter ---');
  produk1.nama = 'TV'; // Gagal
  print('Nama tetap: ${produk1.nama}');

  produk1.harga = -100.0; // Gagal
  print('Harga tetap: Rp ${produk1.harga}');
}

// Nama        : Putra Dhafir Nashrullah

class RekeningBank {
  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening, this._saldo);

  void deposit(double jml) {
    if (jml > 0) {
      _saldo += jml;
      print('Deposit Rp $jml Berhasil.');
    }
  }

  void tarik(double jml) {
    if (jml > 0 && _saldo >= jml) {
      _saldo -= jml;
      print('Tarik Rp $jml Berhasil.');
    } else {
      print('Gagal: Saldo tidak cukup atau input tidak valid.');
    }
  }

  void cekSaldo() {
    print('Rekening: $_nomorRekening | Saldo: Rp $_saldo');
  }
}

void main() {
  var bank = RekeningBank('2514-10002', 2000000);

  bank.cekSaldo();
  bank.tarik(500000);
  bank.cekSaldo();
}

class User {
  String _username;
  String _password;
  String _email;

  User(this._username, this._password, this._email);

  String get username => _username;
  String get email => _email;

  bool login(String username, String password) {
    if (username == _username && password == _password) {
      print("Login berhasil!");
      return true;
    } else {
      print("Login gagal! Username atau password salah.");
      return false;
    }
  }

  void gantiPassword(String passwordLama, String passwordBaru) {
    if (passwordLama == _password) {
      _password = passwordBaru;
      print("Password berhasil diganti.");
    } else {
      print("Password lama salah.");
    }
  }
}

void main() {
  var user1 = User("admin", "12345", "putra@email.com");

  user1.login("admin", "12345");

  user1.gantiPassword("12345", "67890");

  user1.login("admin", "67890");
}
